class Solution {

    public int maxProfit(int[] prices) {
        int minPrice=prices[0];
        int maxProfit=0;
        for(int i=1; i<prices.length;i++){
             minPrice=Math.min(prices[i],minPrice);
            int currntProfit=prices[i]-minPrice;
             maxProfit=Math.max(currntProfit,maxProfit);

        }
        
            return maxProfit;
    }
}
